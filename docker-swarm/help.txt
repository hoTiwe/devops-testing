# Поднимает постгрес и редис, проверяем статус
docker service create \
  --name postgres \
  --network db-postgres-net \
  -e POSTGRES_PASSWORD=postgres \
  -e POSTGRES_USER=postgres \
  -e POSTGRES_DB=postgres \
  postgres:16

docker service create \
  --name redis \
  --network ds-redis-net \
  redis:7

docker service ls
docker service ps postgres
docker service ps redis

# деплоим стек 
docker stack deploy -c docker-compose.yml mystack

docker ps --filter label=com.docker.swarm.service.name=mystack_ubuntu-client

apt-get update
apt-get install -y postgresql-client redis-tools ca-certificates iputils-ping netcat

psql -h postgres -U postgres -d postgres -c "SELECT 1;"

redis-cli -h redis -p 6379 PING
